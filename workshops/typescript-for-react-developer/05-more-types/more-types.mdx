## Type Alias

```ts
type PhoneFormat = string | number;
```

## Interface

```ts
interface Animal {
  name: string;
}

interface Dog extends Animal {
  woof: () => void;
}
```

## Index signature

```ts
interface Person {
  name: string;
  age: number;
  phone: string;
}

interface PersonById {
  [id: string]: Person;
}
```
